<script src="https://code.jquery.com/jquery-3.7.1.min.js"></script>

<script>
function noJQ(){
  let x = new XMLHttpRequest();
  x.onload = () => out1.value = x.responseText;
  x.open("GET","data.txt");
  x.send();
}

// B. AJAX with jQuery
function withJQ(){
  $.get("data.txt",(d)=> out2.value = d);
}

// C. getJSON()
function getJ(){
  $.getJSON("data.json",(d)=>{
    json1.value = "Name: " + d.name + "\nAge: " + d.age;
  });
}

// D. parseJSON()
function parseJ(){
  let s = '{"city":"Bangalore","pincode":560001}';
  let d = JSON.parse(s);
  json2.value = "City: " + d.city + "\nPincode: " + d.pincode;
}
</script>
